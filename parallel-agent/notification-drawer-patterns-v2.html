🗄 Database Schema 
CREATE TABLE notifications (
  id               UUID         PRIMARY KEY,
  user_id          UUID         REFERENCES users(id),
  type             notification_type_enum,  -- see enum below
  title            TEXT         NOT NULL,
  body             TEXT,
  link_url         TEXT,
  is_pinned        BOOLEAN      DEFAULT false,
  read_at          TIMESTAMPTZ,
  dismissed_at     TIMESTAMPTZ,
  auto_dismiss_at  TIMESTAMPTZ,
  created_at       TIMESTAMPTZ  DEFAULT NOW()
);

CREATE TYPE notification_type_enum AS ENUM (
  'system',
  'student_alert',
  'queue_update',
  'teacher_message',
  'child_activity',
  'school_announcement',
  'p0_incident',
  'p1_incident',
  'feedback_spike',
  'release_gate'
);

-- Unread badge count
SELECT COUNT(*)
FROM   notifications
WHERE  user_id = :id
  AND  read_at IS NULL
  AND  dismissed_at IS NULL;

-- Mark all read
UPDATE notifications
SET    read_at = NOW()
WHERE  user_id = :id
  AND  read_at IS NULL;

🎬 Drawer Animation & JS API 
/* Drawer CSS — closed state */
.notif-drawer {
  position:   fixed;
  top:        0;
  right:      0;
  width:      380px;           /* full-screen on mobile */
  height:     100vh;
  background: #161b22;
  border-left: 1px solid rgba(255,255,255,.08);
  z-index:    500;
  transform:  translateX(100%);
  transition: transform 250ms ease;
}

/* Open state */
.notif-drawer.open {
  transform: translateX(0);
}

/* JS API */
function openDrawer()  { drawer.classList.add('open'); scrim.show(); }
function closeDrawer() { drawer.classList.remove('open'); scrim.hide(); }
function toggleDrawer() { drawer.classList.toggle('open'); }

/* Mobile breakpoint */
@media (max-width: 600px) {
  .notif-drawer { width: 100%; }
}
👥 Persona Rules
  • 👩‍🏫
    Teacher: Receives student alerts, queue updates, messages, system. Push optional. Drawer always available in authenticated session.
  • 👨‍👩‍👧
    Parent: Receives child activity (positive language only), teacher messages (platform-routed, no personal contact), school announcements, system. Child receives NO notifications — parent device receives them.
  • 🔧
    Owner/Ops: Receives all incident types with push. P0 non-dismissible and pinned. All types delivered as both push + in-drawer.
  • 🧒
    Child persona: NO notification drawer at all. Child UI has no bell icon. All child-related notifications go to the linked parent device.
Child persona rule: The child play interface (child-* screens) has no notification system. There is no bell icon, no drawer, and no in-app notification rendering. Push notifications relating to a child's activity (XP earned, skills completed, session end) are delivered exclusively to the parent's registered device via FCM/APNs. The child sees none of them.
