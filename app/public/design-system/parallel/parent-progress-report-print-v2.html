Database Schema
CREATE TABLE parent_report_requests ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), parent_id UUID NOT NULL REFERENCES users(id), child_id UUID NOT NULL REFERENCES children(id), period_start DATE NOT NULL, period_end DATE NOT NULL, sections TEXT[], generated_at TIMESTAMPTZ DEFAULT now(), pdf_url TEXT );
Retention policy: Generated PDFs stored for 30 days, then automatically deleted. Parents may re-generate on request at any time. No permanent PDF storage.
